public class Secondlargest
{
	public static void main(String[] args) {

        int [] arr=new int[]{10,20,32,40,89,50,90};
        int n=arr.length;
        for(int i=0; i<2; i++){
            for(int j=0; j<n-1; j++){
                if(arr[j]>arr[j+1]){
                    int temp=arr[j];
                    arr[j]=arr[j+1];
                    arr[j+1]=temp;
                }
            }
        }
        System.out.println("Second largest element is:"+arr[arr.length-2]);
	}
}
